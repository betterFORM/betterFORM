Description

This form control allows the user to make multiple selections from a set of choices..

select1 Control with appearance 'minimal'

Default rendering of a select1 if no appearance is given.

XForms Markup

<!--XForms UI-->
<xf:select1 ref="item/value">
    <xf:label>minimal select1</xf:label>
    <xf:itemset nodeset="instance('codes')/country">
        <xf:label ref="country-name"></xf:label>
        <xf:value ref="code"></xf:value>
    </xf:itemset>
</xf:select1>

open select1 Control

Select1 with @selection='open' allows the user to input a value that does not exist in the list of choices. Simply type into the input field.

XForms Markup

<!--XForms UI-->
<xf:select1 id="select12" ref="item/value" selection="open">
    <xf:label>selection open</xf:label>
    <xf:choices>
        <xf:item>
            <xf:label>AFGHANISTAN</xf:label>
            <xf:value>AF</xf:value>
        </xf:item>
        <xf:item>
            <xf:label>LAND ISLANDS</xf:label>
            <xf:value>AX</xf:value>
        </xf:item>
        <xf:item>
            <xf:label>ALBANIA</xf:label>
            <xf:value>AL</xf:value>
        </xf:item>
        <xf:item>
            <xf:label>ALGERIA</xf:label>
            <xf:value>DZ</xf:value>
        </xf:item>
    </xf:choices>
</xf:select1>
